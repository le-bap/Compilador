sirva("Quantas receitas voce deseja ver?");
ingrediente total;
prove("%i" total);

bata (i = 1; i <= total) {
    sirva("Escolha uma opcao:");
    sirva("1 - Bolo");
    sirva("2 - Salada");
    sirva("3 - Sair");
    ingrediente opcao;
    prove("%i" opcao);

    deguste (opcao == 1) {
        sirva("Voce escolheu BOLO!");
        sirva("Quantas fatias deseja?");
        ingrediente fatias;
        prove("%i" fatias);

        cozinhe_enquanto (fatias > 0) {
            sirva("Servindo uma fatia...");
            fatias = fatias - 1;

            deguste (fatias == 0){
                sirva("Acabaram as fatias!");
            }
        }
    }

    tempere {
        deguste (opcao == 2) {
            sirva("Voce escolheu SALADA!");
            sirva("Deseja adicionar molho? (sim ou nao)");
            receitinha resposta;
            prove("%r" resposta);

            deguste (resposta == "sim") {
                sirva("Molho adicionado!");
            }
            tempere {
                sirva("Sem molho, ok.");
            }
        }

        tempere {
            deguste (opcao == 3) {
                sirva("Encerrando o cardapio...");
                parar;
            }

            tempere {
                sirva("Opcao invalida, tente novamente.");
            }
        }
    }
}

sirva("Agora vamos contar ate 5 usando WHILE:");
ingrediente contador = 0;

cozinhe_enquanto (contador <= 5) {
    sirva({contador});
    contador = contador + 1;
}
